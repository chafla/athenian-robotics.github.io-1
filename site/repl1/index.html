for i in range(10):

        print(str(i) + " is even")
    else:
        print(str(i) + " is odd")

Functions

def is_even(val):
    return val%2 == 0

def is_odd(val):
    return val%2 != 0

for i in range(10):
    if is_even(i):
        print(str(i) + " is even")
    if is_odd(i):
        print(str(i) + " is odd")
